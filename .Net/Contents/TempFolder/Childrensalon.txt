
Select TR_documents.* from TR_Orders inner join CL_Clients on TR_orders.idClient = CL_Clients.idClient
inner join TR_Jobs on TR_Jobs.idOrder = TR_Orders.idOrder
Inner join TR_Translations on TR_Translations.idJob = TR_Jobs.idJob
inner join TR_Quotes on TR_Quotes.idQuote = TR_Jobs.idQuote
Inner join TR_documents on TR_documents.idDocument = TR_Jobs.idOriginalDoc
inner join _WorkflowSteps on _WorkflowSteps.idTranslation = TR_Translations.idTranslation and _WorkflowSteps.idTask = 10 
and _WorkflowSteps.status = 1
where CL_Clients.idAccount = 34953 and dateCreated >= '2023-01-01' and dateCreated <= '2023-03-31'
and TR_Quotes.ToLanguage = 31
